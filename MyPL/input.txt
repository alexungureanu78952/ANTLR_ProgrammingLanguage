// ===== SEMANTIC ERRORS =====

// 5. Eroare: Unicitatea variabilelor globale - două variabile cu același nume
int globalVar = 10;
int globalVar = 20;  // Eroare: variabilă globală duplicată

// 6. Eroare: Unicitatea funcțiilor - două funcții cu același nume și parametri
int addNumbers(int a, int b) {
    return a + b;
}

int addNumbers(int a, int b) {  // Eroare: funcție duplicată
    return a + b + 1;
}

// 7. Eroare: Apelarea unei funcții nedefinite
int testUndefinedCall() {
    int result = undefinedFunction(5);  // Eroare: funcție nedefinită
    return result;
}

// 8. Eroare: Lipsă funcție main SAU funcții main multiple
void main() {
    int x = 10;
}

void main() {  // Eroare: a doua funcție main
    int y = 20;
}

// 9. Eroare: Recursivitatea pentru funcția main
void callMain() {
    main();  // Eroare: apelare funcție main din altă funcție
}

void anotherMain() {
    main();  // Eroare: apelare funcție main
}

// 10. Eroare: Utilizarea variabilelor înainte de declarare
int useBeforeDeclaration() {
    int result = undeclaredVar + 5;  // Eroare: variabilă nedeclarată
    int undeclaredVar = 10;
    return result;
}

// 11. Eroare: Unicitatea variabilelor locale - două variabile locale cu același nume
int duplicateLocalVars() {
    int localVar = 5;
    int localVar = 10;  // Eroare: variabilă locală duplicată
    return localVar;
}

// 12. Eroare: Conflict între variabilă locală și parametru
int conflictParamLocal(int param1) {
    int param1 = 20;  // Eroare: variabilă locală cu același nume ca parametrul
    return param1;
}

// 13. Eroare: Incompatibilitate tipuri la inițializare
int wrongTypeInit() {
    int number = "some text";  // Eroare: int nu poate fi inițializat cu string
    float value = "another string";  // Eroare: float nu poate fi inițializat cu string
    string text = 42;  // Eroare: string nu poate fi inițializat cu int
    return 0;
}

// Exemplu corect pentru comparație
double correctInit() {
    double PI = 3.14;  // Corect: double cu float literal
    return PI;
}

// 14. Funcție pentru testarea erorilor de apel
int functionWithParams(int a, float b, string c) {
    return a;
}

// 15. Eroare: Număr incorect de argumente
int wrongArgumentCount() {
    int result1 = functionWithParams(10);  // Eroare: prea puține argumente
    int result2 = functionWithParams(10, 3.5, "text", 99);  // Eroare: prea multe argumente
    return result1;
}

// 16. Eroare: Tipuri incompatibile la apelul funcției
int wrongArgumentTypes() {
    int result = functionWithParams("string", "string", 10);  // Eroare: tipuri greșite
    return result;
}

// 17. Eroare: Incompatibilitate tip valoare returnată
int wrongReturnType() {
    return "this is a string";  // Eroare: returnează string în loc de int
}

float anotherWrongReturn() {
    return "another string";  // Eroare: returnează string în loc de float
}

// 18. Eroare: Lipsă instrucțiune return pentru funcții non-void
int missingReturn(int x) {
    int y = x + 5;
    // Eroare: lipsește return
}

// Funcție corectă cu return
int correctReturn(int x) {
    return x * 2;
}

// 19. Eroare: Modificarea variabilelor constante
void modifyConstant() {
    const int CONSTANT_VALUE = 100;
    CONSTANT_VALUE = 200;  // Eroare: nu poți modifica o constantă
}

const int GLOBAL_CONSTANT = 50;

void modifyGlobalConstant() {
    GLOBAL_CONSTANT = 75;  // Eroare: nu poți modifica o constantă globală
}

// 20. Eroare: Operatori de asignare pe constante
void assignOperatorsOnConst() {
    const int counter = 10;
    counter += 5;  // Eroare: operator de asignare pe constantă
    counter++;     // Eroare: increment pe constantă
}

// ===== SYNTAX ERRORS =====

// 21. Diverse erori de sintaxă

// Lipsă punct și virgulă
int missingSemicolon() {
    int x = 10  // Eroare: lipsește ;
    return x;
}

// Paranteze nepotrivite
int mismatchedParentheses() {
    int x = (5 + 3;  // Eroare: lipsește )
    return x;
}

// Acolade nepotrivite
int mismatchedBraces() {
    int x = 10;
    return x;
// Eroare: lipsește }

// Declarație invalidă
int invalidDeclaration() {
    x = 10;  // Eroare: lipsește tipul
    return x;
}

// Expresie invalidă
int invalidExpression() {
    int x = 5 + + 3;  // Eroare: operatori multipli
    return x;
}
